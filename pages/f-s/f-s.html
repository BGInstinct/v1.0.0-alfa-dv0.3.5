Ensuring the security of your files is crucial to protect sensitive information from unauthorized access. Below, you will find example code for a simple encryption program written in both Python and C#. Additionally, there is a more complex example available to demonstrate advanced encryption techniques for robust file security.

Pyton

import random import string chars = string.punctuation + string.digits + string.ascii_letters chars = list(chars) key = chars.copy() random.shuffle(key) #ENCRYPT plain_text = input("Enter a message to encrypt: ") cipher_text = "" for letter in plain_text: index = chars.index(letter) cipher_text += key[index] print(f"original message : {plain_text}") print(f"encrypted message: {cipher_text}") #DECRYPT cipher_text = input("Enter a message to encrypt: ") plain_text = "" for letter in cipher_text: index = key.index(letter) plain_text += chars[index] print(f"encrypted message: {cipher_text}") print(f"original message : {plain_text}")

C#

#NOTE: REMOVE THE SLASH FROM <\char> using System; using System.Collections.Generic; using System.Linq; class Program { static void Main() { // Initialize character sets string chars = " " + "!\"#$%&'()*+,-./:;<=>?@[\\]^_`{|}~" + "0123456789" + "ABCDEFGHIJKLMNOPQRSTUVWXYZ" + "abcdefghijklmnopqrstuvwxyz"; List<\char> charList = chars.ToList(); List<\char> key = new List<\char>(charList); // Shuffle key Random rng = new Random(); int n = key.Count; while (n > 1) { n--; int k = rng.Next(n + 1); char value = key[k]; key[k] = key[n]; key[n] = value; } foreach (char c in key) { Console.WriteLine(c); } foreach (char a in charList) { Console.WriteLine(a); } // ENCRYPT Console.Write("Enter a message to encrypt: "); string plainText = Console.ReadLine(); string cipherText = ""; foreach (char letter in plainText) { int index = charList.IndexOf(letter); cipherText += key[index]; } Console.WriteLine($"Original message: {plainText}"); Console.WriteLine($"Encrypted message: {cipherText}"); // DECRYPT Console.Write("Enter a message to decrypt: "); cipherText = Console.ReadLine(); plainText = ""; foreach (char letter in cipherText) { int index = key.IndexOf(letter); plainText += charList[index]; } Console.WriteLine($"Encrypted message: {cipherText}"); Console.WriteLine($"Original message: {plainText}"); } }
